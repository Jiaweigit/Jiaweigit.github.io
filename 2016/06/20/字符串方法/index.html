字符串方法

2016-06-20 字符串方法 前端

charAt() 在该字符串中根据下标（索引）来找对应的字符

   var str = "123456";
var result = str.charAt(2);
    console.log(result); //3
    console.log(str);    //123456

charCodeAt() 在该字符串中根据下标找对应字符的字符编码

var str = "123456"
var result = str.charCodeAt(2);
console.log(result);    //51 
console.log(str);    //123456

toUpperCase() 将小写转换成大写

var str ="Ab你cfdgCC好";
var result = str.toUpperCase();
console.log(result);    //AB你CFDGCC好
console.log(str);    //Ab你cfdgCC好

toLowerCase() 将大写字母转小写

var str ="Ab你cfdgCC好";
var result = str.toLowerCase();//将大写字母转小写
console.log(result);    //ab你cfdgcc好
console.log(str);    //Ab你cfdgCC好

slice() 提取字符串的某个部分，并以新的字符串返回被提取的部分

参数 :开始截取位置,结束截取位置(不包含结束) 不支持颠倒顺序 支持负数
var str = "Web前端开发工程师";    
var newStr = str.slice(4,-1);
console.log(newStr); //端开发工程
console.log(str);    //Web前端开发工程师

substring() 提取字符串中介于两个指定下标之间的字符

参数 :开始截取位置,结束截取位置(不包含结束)比要提取的子串的最后一个字符在stringObject中的位置多1 支持颠倒顺序 不支持负数(当成0处理)其内容是从 start 处到 stop-1 处的所有字符
var str = "Web前端开发工程师";    
var str2 = str.substring(4,0);
console.log(str2);    //Web前
console.log(str);    //Web前端开发工程师

indexOf() 返回某个指定的字符串值在字符串中首次出现的位置

参数： 第一个参数  查找的字符串  第二参数（第二参数不写 默认为0） 开始查找的位置
如果要检索的字符串值没有出现，则该方法返回 -1。
var str = "www.baidu.com/index.html";
var result = str.indexOf("i",15);
console.log( result );    //-1

lastIndexOf() 返回一个指定的字符串值最后出现的位置，在一个字符串中的指定位置从后向前搜索

var str = "www.baidu.com/index.html";
var result = str.lastIndexOf("i",15);
console.log(result);    //14

fromCharCode() 可接受一个指定的 Unicode 值，然后返回一个字符串

var result = String.fromCharCode(25771);
console.log(result);    //撫

concat() 方法用于连接两个或多个字符串。

var str1 = "hello"
var str2 = "world"
var result = str1.concat(str2);
console.log( result );    //helloworld

replace() 方法用于在字符串中用一些字符替换另一些字符，或替换一个与正则表达式匹配的子串。

var str = "hello";
var result = str.replace(/o/,"oworld");
console.log( result );    //helloworld

split() 方法用于把一个字符串分割成字符串数组

var str = "hello world";
var result = str.split(" ");
console.log( result );    //["hello", "world"]

valueOf() 方法可返回 String 对象的原始值

var str = "hello world";
var result = str.valueOf();
console.log( result );    //hello world

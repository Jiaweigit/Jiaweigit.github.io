正则表达式

正则表达式

声明和测试方法

1.构造函数法;
var reg1 = new RegExp(/abc/);
console.dir(reg1); // /abc/
console.log(reg1); // /abc/
console.log(typeof reg1); // object

2.直接量;字面量;
var reg2 = /abc/;
var reg3 = /abc/;
console.dir(reg2); // /abc/
console.log(reg2); // /abc/
console.log(typeof reg2); // object

简单数据类型比较的时候,比较的是数值;
复杂数据类型比较的时候,比较的是地址值;
console.log(reg1 === reg2); //false
console.log(reg3 === reg2); //false

正则表达式的使用

test(); 测试参数是否符合正则表达式的约束; 符合就是true; 不符合就是false;
var reg1 = new RegExp(/abc/);
console.log(reg1.test(“abc”)); //true
console.log(reg1.test(“def”)); //false

五大内部类

  1. 预定义类:正则表达式预先设计好的一些元字符
    (1) . 查找单个字符，除了换行和回车以外;
    console.log(/./.test(“ “)); //true
    console.log(/./.test(“nmv”)); //true
    console.log(/./.test(“123”)); //true
    console.log(/./.test(“#$%^”)); //true
    换行和回车不行
    console.log(/./.test(“\r”)); //false
    console.log(/./.test(“\n”)); //false
    console.log(/./.test(“\n\r”)); //false
    console.log(/./.test(“”)); //false
    console.log(“a\r\na”);
    (2) \d 数字字符; \D 非数字;
    console.log(/\d/.test(“1233456”)); //true
    console.log(/\d/.test(“abc #%$^”)); //false
    console.log(/\D/.test(“1233456”)); //false
    console.log(/\D/.test(“abc #%$^”)); //true
    (3) \s 空白字符; \S 非空白字符;
    console.log(/\s/.test(“ “)); //true
    console.log(/\S/.test(“ “)); //false
    (4) \w 单词字符; \W 非单词字符;
    console.log(/\w/.test(“aA0“)); //true
    console.log(/\w/.test(“!@#$%^”)); //false
    console.log(/\W/.test(“aA0    “)); //false
    console.log(/\W/.test(“!@#$%^”)); //true

  2. 简单类
    (1)直接书写; (必须是完整的，只多不能少,顺序必须对)
    console.log(/abc/.test(“abc”)); //true
    console.log(/abc/.test(“xxabcxx”)); //true
    console.log(/abc/.test(“acb”)); //false
    console.log(/abc/.test(“ab”)); //false
    console.log(/abc/.test(“abxxxxc”)); //false
    (2)写到[]里; (只要包含里面的任何一个就可以)
    console.log(/[abc]/.test(“axxxxxxx”)); //true
    console.log(/[abc]/.test(“vbxxxxxxx”)); //true
    console.log(/[abc]/.test(“vyyyucxxxxxxx”)); //true
    console.log(/[abc]/.test(“vyyyuxxxxxxx”)); //false

  3. 负向类: 把^写到[]里面的最前面就是负向类;
    被测试的参数只要有正则中没有规定的就是true;
    如果完整包含,或者只是约束的一部分;那么就是fales;
    console.log(/[^abc]/.test(“abccr”)); //true
    console.log(/[^abc]/.test(“xxx”)); //true
    console.log(/[^abc]/.test(“abx”)); //true
    console.log(/[^abc]/.test(“abc”)); //false
    console.log(/[^abc]/.test(“ab”)); //false

  4. 范围类: - ; 从哪到哪的意思;一般与[]连用;
    console.log(/[a-d]/.test(“cc”)); //true
    console.log(/[a-d]/.test(“ecc”)); //true
    console.log(/[a-d]/.test(“e”)); //false
    console.log(/[0-9]/.test(“e”)); //false
    console.log(/[A-F]/.test(“e”)); //false

  5. 组合类: 就是把范围类,一起用;
    console.log(/[a-dA-D0-5]/.test(“cC3”)); //true

正则边界: 1.以什么为开始; 2.以什么为结束; 3.必须是这样,不能多也不能少;

^: 1.[]里面的负向类;    2.  进入//里面直接书写的^;以什么...开始;
$: 以什么...为结束;
想输入元素组就要用\符;
  1. 以abc为开始;
    console.log(/^abc/.test(“a”)); //false
    console.log(/^abc/.test(“ab”)); //false
    console.log(/^abc/.test(“abc”)); //true
    console.log(/^abc/.test(“abcd”)); //true
    console.log(/^abc/.test(“abcde”)); //true
    console.log(/^abc/.test(“acbxxx”)); //false
  2. 以abc为结束;
    console.log(/abc$/.test(“a”)); //false
    console.log(/abc$/.test(“ab”)); //false
    console.log(/abc$/.test(“abc”)); //true
    console.log(/abc$/.test(“xxxxabc”)); //true
    console.log(/abc$/.test(“abcde”)); //false
    console.log(/abc$/.test(“acbxxx”)); //false
    3.必须是这样,不能多也不能少;
    console.log(/^abc$/.test(“ab”)); //false
    console.log(/^abc$/.test(“abc”)); //true
    console.log(/^abc$/.test(“abcd”)); //false
    console.log(/^abc$/.test(“abcabc”)); //false

正则表达式-量词

量词四种:
1.*     重复零次或更多    (>=0)

2.+ 重复一次或更多次 (>=1)
3.? 重复零次或一次 (0||1) 要么有 要么没有
4.{} 重复多少次的意思

    • 重复零次或更多 (>=0)
      a : 代表a不出现也可以,出现也可以, 多少次都行;
      console.log(/^a      $/.test(“”)); //true
      console.log(/^a$/.test(“a”)); //true
      console.log(/^a      $/.test(“aa”)); //true
      console.log(/^a$/.test(“aaa”)); //true
      console.log(/^a      $/.test(“aaaa”)); //true
      console.log(/^a*$/.test(“aaaab”)); //false

    • 重复一次或更多次 (>=1)
      a* : 代表a出现至少一次;
      console.log(/^a+$/.test(“”)); //false
      console.log(/^a+$/.test(“a”)); //true
      console.log(/^a+$/.test(“aa”)); //true
      console.log(/^a+$/.test(“aaa”)); //true
      console.log(/^a+$/.test(“aaaa”)); //true
      console.log(/^a+$/.test(“aaaab”)); //false

      1. ? 重复零次或一次 (0||1) 要么有 要么没有
        a* : 代表a出现零次或一次都行;
        console.log(/^a?$/.test(“”)); //true
        console.log(/^a?$/.test(“a”)); //true
        console.log(/^a?$/.test(“aa”)); //false
        console.log(/^a?$/.test(“aaa”)); //false
        console.log(/^a?$/.test(“aaaa”)); //false
        console.log(/^a?$/.test(“aaaab”)); //false

4.{} 重复多少次的意思
(1). {4} 重复4次; (2). {4,} 重复4次以上; (3). {4,8} 重复4-8次;
{n} n次 （x=n）
{n,} 重复n次或更多 (x>=n)
{n,m} 重复出现的次数比n多但比m少 (n<=x<=m)

console.log(/^a{3}$/.test(“aa”)); //false
console.log(/^a{3}$/.test(“aaa”)); //true
console.log(/^a{3}$/.test(“aaaa”)); //false
console.log(/^a{3}$/.test(“aaaab”)); //false

console.log(/^a{3,}$/.test(“aa”)); //false
console.log(/^a{3,}$/.test(“aaa”)); //true
console.log(/^a{3,}$/.test(“aaaa”)); //true
console.log(/^a{3,}$/.test(“aaaab”)); //false

console.log(/^a{3,5}$/.test(“aa”)); //false
console.log(/^a{3,5}$/.test(“aaa”)); //true
console.log(/^a{3,5}$/.test(“aaaa”)); //true
console.log(/^a{3,5}$/.test(“aaaaa”)); //true
console.log(/^a{3,5}$/.test(“aaaaaa”)); //false
console.log(/^a{3,5}$/.test(“aaaaaaa”)); //false
console.log(/^a{3,5}$/.test(“aaaaaaab”)); //false

正则表达式-优先级

1.（） 2.| 3.^$ 4.量词
| 是或的意思;在正则中没有||; 满足一个添加就行;

正则表达式-三个函数

1.replace(); 替换字符串;
var str = “aaaBBBAAACCCaaa”;
console.log(str.replace(“aaa”,”ooo”)); //oooBBBAAACCCaaa
console.log(str.replace(/aaa/g,”ooo”)); //oooBBBAAACCCooo
console.log(str.replace(/aaa/gi,”ooo”)); //oooBBBoooCCCooo
i: 忽略大小写; g:全局替换;

2.trim(); 去除字符串两边的空格; ie678不支持;
var str = “ aaa bbb ccc “;
console.log(str); // aaa bbb ccc
console.log(str.trim()); //aaa bbb ccc

3.search(正则); 给字符查索引值;
var str = “abcdefg”;
console.log(str.search(/cd/)); //2

正则表达式-trim()封装

function trim(str){
return str.replace(/(^\s+) | (\s+$) /g,””);
}
var str = “ aaa bbb ccc “;
console.log(str) // aaa bbb ccc
console.log(trim(str)) //aaa bbb ccc
